Simple Integration

main.cpp 

#include "hpq/writer.h"

int main() {
    // 1. Define Schema
    hpq::Schema schema;
    schema.AddColumn("id", hpq::Type::INT64);
    schema.AddColumn("score", hpq::Type::FLOAT);

    // 2. Configure Options
    hpq::WriterOptions options;
    options.use_gpu_compression = true; // Auto-fallback supported

    // 3. Write Data
    hpq::ParquetWriter writer("data.parquet", options);
    writer.Init(schema);

    std::vector ids = {1, 2, 3, 4, 5};
    writer.WriteColumn(0, ids.data(), ids.size());

    writer.Close();
    return 0;
}
